3.3.	Write a JavaScript program to compute the sum of the given two integers. If the two values are same, then returns triple their sum.

<!doctype html>
<html>
<head>

</head>
<body>
    <script>
        function add()
        {
          var num1, num2, sum;
          num1 = parseInt(document.getElementById("first").value);
          num2 = parseInt(document.getElementById("second").value);
        
          if(num1 == num2)
          {
            sum = 3 * (num1 + num2);
            document.getElementById("answer").value = sum;
          }
          else
          {
            sum = num1 + num2;
            document.getElementById("answer").value = sum;
          }
        }
    </script>
<p>First Number: <input id="first"></p>
<p></p>Second Number: <input id="second"></p>
<button onclick="add()">Add</button>
<p>Sum = <input id="answer"></p>

</body>
